  unsigned char *p1 = p;

    p1++;
  }
}


int main (void) {
  int i,j;
  short a[2][3];
  int b[2];

  for (i=0;i<2;i++) {
    b[i] = i;
    for (j=0;j<3;j++)
      a[i][j] = 3*i+j;
  }

  printf ("b: \n");
  dump (b, sizeof(b));
  printf ("a: \n");
  dump (a, sizeof(a));

  return 0;
}

tem um buraco porque para que o endereço ocupe a posição de memória,
a posição precisa ser múltipla da quantidade de bytes do tipo que será alocado.


3. 

a.
X1 é armazenado na memória com 1 byte de c1 (char) seguido de 3 bytes depois 4 bytes de i (int) 
por fim mais 1 byte de c2 (char).

b.
X2 é armazenado na memória com 8 bytes de l (long) seguido de 1 byte de c (char), no entanto, a struct inteira deve
ter tamanho múltiplo do maior alinhamento que nesse caso é 8 bytes, se ele tem 9 bytes no total (long + char), logo,
ele precisa de mais 7 bytes para ser um múltiplo de 8, logo X2 tem 16 bytes.

c.
X3 é armazenado na memória com 4 bytes de i seguido de 1 byte de c1 (char) e depois mais 1 byte de c2 (char)
que dão 6 bytes no total e para alcançar o tamanho múltiplo do maior alinhamento (4 bytes) ele precisa de mais 2 bytes,
resultando em 8 bytes no total.

d. 
X4 é armazenado na memória com 16 bytes de X2 seguido de 1 byte de c (char) resultando em 17 bytes, apesar de X2 ter 16 bytes
ele continua tendo como maior alinhamento 8 bytes (long), logo, o maior alinhamento de X4 também é 8 bytes, então seu tamanho final
precisa ser múltiplo de 8, e o próximo múltiplo de 8 depois de 16 é 24, ele precisa de mais 7 bytes para alcançar os 24 bytes.

e.
X5 é armazenado na memória com 3 bytes de c1,c2 e c3 (char).

f.
X6 é armazenado na memória com 2 bytes de s1 (short), pula duas posições porque 2 e 3 não são múltiplos de
4, depois ele aloca os 4 bytes de i (int), em seguida ele armazena 3 bytes de c[3] (char), depois pula novamente 1 espaço
(porque 11 não é múltiplo de 2) e por fim aloca 2 bytes de s2 (short) e mais 2 bytes para que o tamanho seja múltiplo de 4 bytes, 
no total tem 16 bytes.

g.
Como a union guarda o valor do maior alinhamento, apesar do maior tamanho ser char c[u] que tem 5 bytes 
o maior é alinhamento é 4 bytes (int), então o tamanho final precisa ser múltiplo de 4 que é 8. 

h.
Novamente, apesar do maior tamanho ser de char c[5] (5 bytes) o maior alinhamento é 2 bytes de short o próximo múltiplo é 4 .
